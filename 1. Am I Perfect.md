#**1. Am I Perfect?**


```
function perfectNum(num){
    var sum = 0,i=0;
    while(i!=num) {
        if(num%i == 0)
            sum +=i;
        i++;
    }
    if(sum == num)
        return "Perfect";
    else if(sum > num)
        return "Abundent";
    else if(sum < num)
        return "Deficient";
}
```
