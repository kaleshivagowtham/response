**1. Greater than and less than in a matrix**
```
function elements(matrix) {
    let list = [],i=0,j=0,rows=matrix.length,cols=matrix[0].length;
    while(i<rows) {
        j=0;
        while(j<cols) {
            let curr = matrix[i][j],canAdd=true;
            let curri=i,currj=0;
            while(currj<cols) {
                if(matrix[curri][currj] > curr){
                    canAdd = false;
                    break;
                }
                currj++;
            }
            curri=0;
            currj=j;
            while(curri<rows) {
                if(matrix[curri][currj] < curr){
                    canAdd=false;
                    break;
                }
                curri++;
            }
            if(canAdd === true)
                list.push(curr);
            j++;
        }
        i++;
    }
    return list;
}
```
