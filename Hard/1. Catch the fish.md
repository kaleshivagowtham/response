#**1. Catch the fish**

```
function escapedFishes() {
    
    let k=prompt("K :");
    let l=prompt("L :");
    let m=prompt("M :");
    let n=prompt("N :");
    let total=prompt("Total :");
    let i=1,count=0;

    while(i<=total) {
        if(i%k===0||i%l===0||i%m===0||i%n===0)
            count++;
        i++;
    }
    return count;
}
```
