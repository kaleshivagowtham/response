**3. How many trails to 1?**

```

function numberOfSteps(num) {

    if(num == 0)
        return "infinite";

    let count = 0;
    while(num != 1) {
        if(num%2 === 0) {
            num /= 2;
        }
        else {
            num = (3*num)+1;
        }
        count++;
    }
    return count;
}
```
